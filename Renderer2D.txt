        auto TranscodeUtf8ToWide(const std::string_view utf8) -> std::wstring
        {
            if (utf8.empty())
            {
                return {};
            }

            int desired = ::MultiByteToWideChar(CP_UTF8, 0, utf8.data(), -1, nullptr, 0);
            if (desired == 0)
            {
                return {};
            }

            std::wstring result(desired, 0);
            ::MultiByteToWideChar(CP_UTF8, 0, utf8.data(), -1, &result[0], desired);

            result.resize(desired - 1);
            return result;
        }
    } // namespace

    WicImageDecoder::WicImageDecoder(const std::string_view path) : m_Path(path)
    {
        auto factory = wil::CoCreateInstance<IWICImagingFactory>(CLSID_WICImagingFactory);

        // 繝・さ繝ｼ繝繝ｼ繧剃ｽ懈・縺励※縲∫判蜒上ヵ繧｡繧､繝ｫ繧定ｪｭ縺ｿ霎ｼ繧
        THROW_IF_FAILED(factory->CreateDecoderFromFilename(TranscodeUtf8ToWide(path).c_str(), nullptr, GENERIC_READ, WICDecodeMetadataCacheOnDemand, &m_Decoder)
        );
    }

    auto WicImageDecoder::Decode(std::function<std::span<std::byte>(std::size_t)> allocate) -> Pixels::Buffer
    {
        // 逕ｻ蜒上・譛蛻昴・繝輔Ξ繝ｼ繝繧貞叙蠕励☆繧・
        wil::com_ptr<IWICBitmapFrameDecode> frame;
        THROW_IF_FAILED(m_Decoder->GetFrame(0, &frame));

        // 繝輔か繝ｼ繝槭ャ繝医さ繝ｳ繝舌・繧ｿ繝ｼ繧剃ｽ懈・縺励※縲√ヵ繝ｬ繝ｼ繝繧・2bppPBGRA蠖｢蠑上↓螟画鋤縺吶ｋ
        auto factory = wil::CoCreateInstance<IWICImagingFactory>(CLSID_WICImagingFactory);
        wil::com_ptr<IWICFormatConverter> converter;
        THROW_IF_FAILED(factory->CreateFormatConverter(&converter));

        // 32bppPBGRA蠖｢蠑上↓螟画鋤縺吶ｋ
        THROW_IF_FAILED(converter->Initialize(frame.get(), GUID_WICPixelFormat32bppPBGRA, WICBitmapDitherTypeNone, nullptr, 0.0f, WICBitmapPaletteTypeCustom));

        // 逕ｻ蜒上・蟷・・ｫ倥＆縲√ヴ繝・メ繧貞叙蠕励＠縺ｦ縲∝ｿ・ｦ√↑繝舌う繝域焚繧定ｨ育ｮ励☆繧・
        const std::uint32_t width          = GetWidth();
        const std::uint32_t height         = GetHeight();
        const std::uint32_t pitch          = GetPitch();
        const std::size_t totalBytesNeeded = static_cast<std::size_t>(pitch) * height;

        // 繝舌ャ繝輔ぃ繧貞牡繧雁ｽ薙※繧・
        auto target = allocate(totalBytesNeeded);

        if (target.empty() || target.size() < totalBytesNeeded)
        {
            return {};
        }

        // 螟画鋤縺輔ｌ縺溘ヴ繧ｯ繧ｻ繝ｫ繝・・繧ｿ繧偵ヰ繝・ヵ繧｡縺ｫ繧ｳ繝斐・縺吶ｋ
        THROW_IF_FAILED(converter->CopyPixels(nullptr, pitch, static_cast<UINT>(target.size()), reinterpret_cast<BYTE*>(target.data())));

        // 繝舌ャ繝輔ぃ繧定ｿ斐☆
        return {
            .Bytes  = target.data(),
            .Size   = totalBytesNeeded,
            .Width  = GetWidth(),
            .Height = GetHeight(),
            .Pitch  = pitch,
        };
    }

    auto WicImageDecoder::GetWidth() const -> std::uint32_t
    {
        wil::com_ptr<IWICBitmapFrameDecode> frame;
        m_Decoder->GetFrame(0, &frame);

        UINT width  = 0;
        UINT height = 0;

        frame->GetSize(&width, &height);
        return width;
    }

    auto WicImageDecoder::GetHeight() const -> std::uint32_t
    {
        wil::com_ptr<IWICBitmapFrameDecode> frame;
        m_Decoder->GetFrame(0, &frame);

        UINT width  = 0;
        UINT height = 0;

        frame->GetSize(&width, &height);
        return height;
    }

    auto WicImageDecoder::GetPitch() const -> std::uint32_t
    {
        return GetWidth() * 4;
    }

    auto WicImageDecoder::GetPath() const -> std::string_view
    {
        return m_Path;
    }

    auto Cache::Get(ResourceHandle handle) -> wil::com_ptr<ID2D1Bitmap1>
    {
        const auto key = handle.ID;

        if (m_Bitmaps.find(key) == m_Bitmaps.end())
        {
            return nullptr;
        }

        return m_Bitmaps[key];
    }

    auto Cache::GetOrCreateBitmap(ResourceHandle handle, const Pixels::Buffer& pixels) -> wil::com_ptr<ID2D1Bitmap1>
    {
        const auto key = handle.ID;

        if (m_Bitmaps.find(key) == m_Bitmaps.end())
        {
#ifdef _DEBUG
            auto log = std::format("[Renderer2D] <Device::Resource::Cache>: CreateBitmap for ResourceID={}", static_cast<std::uint64_t>(handle.ID));
            Engine::GetInstance().GetDiagnosticsSink().AddEntry(Diagnostics::Entry{ Diagnostics::Severity::Verbose, log });
#endif
            auto bitmap = m_DeviceContext.CreateBitmap(pixels);

            if (!bitmap)
            {
                return nullptr;
            }

            m_Bitmaps[key] = bitmap;
        }

        return m_Bitmaps[key];
    }

    auto Cache::GetOrCreateBrush(const Renderer2D::ColorF color) -> wil::com_ptr<ID2D1SolidColorBrush>
    {
        // clang-format off
        const uint32_t key = (static_cast<uint32_t>(color.Red   * 255) << 24) |
                             (static_cast<uint32_t>(color.Green * 255) << 16) |
                             (static_cast<uint32_t>(color.Blue  * 255) <<  8) |
                             (static_cast<uint32_t>(color.Alpha * 255));
        // clang-format on

        if (m_Brushes.find(key) == m_Brushes.end())
        {
#ifdef _DEBUG
            auto log = std::format("[Renderer2D] <Device::Resource::Cache>: CreateBrush for Color={}", key);
            Engine::GetInstance().GetDiagnosticsSink().AddEntry(Diagnostics::Entry{ Diagnostics::Severity::Verbose, log });
#endif
            auto brush = m_DeviceContext.CreateSolidColorBrush(color);

            if (!brush)
            {
                return nullptr;
            }

            m_Brushes[key] = brush;
        }

        return m_Brushes[key];
    }

    auto Cache::GetOrCreateTextFormat(std::string_view fontName, float fontSize) -> wil::com_ptr<IDWriteTextFormat>
    {
        const auto key = TranscodeUtf8ToWide(fontName) + L"_" + std::to_wstring(fontSize);

        if (m_TextFormats.find(key) == m_TextFormats.end())
        {
#ifdef _DEBUG
            auto log = std::format("[Renderer2D] <Device::Resource::Cache>: CreateTextFormat for Key={}", fontName);
            Engine::GetInstance().GetDiagnosticsSink().AddEntry(Diagnostics::Entry{ Diagnostics::Severity::Verbose, log });
#endif
            auto textFormat = m_DeviceContext.CreateTextFormat(fontName, fontSize);

            if (!textFormat)
            {
                return nullptr;
            }

            m_TextFormats[key] = textFormat;
        }

        return m_TextFormats[key];
    }

    auto Cache::GetOrCreateTextLayout(const std::string_view text, wil::com_ptr<IDWriteTextFormat> textFormat) -> wil::com_ptr<IDWriteTextLayout>
    {
        const auto key = TranscodeUtf8ToWide(text) + L"_" + std::to_wstring(reinterpret_cast<uintptr_t>(textFormat.get()));

        if (m_TextLayouts.find(key) == m_TextLayouts.end())
        {
#ifdef _DEBUG
            auto log = std::format("[Renderer2D] <Device::Resource::Cache>: CreateTextLayout for Key={}", text);
            Engine::GetInstance().GetDiagnosticsSink().AddEntry(Diagnostics::Entry{ Diagnostics::Severity::Verbose, log });
#endif
            auto textLayout = m_DeviceContext.CreateTextLayout(text, textFormat);

            if (!textLayout)
            {
                return nullptr;
            }

            m_TextLayouts[key] = textLayout;
        }

        return m_TextLayouts[key];
    }

    RenderTarget::RenderTarget(Device::Context& context, HWND hwnd) : m_TargetWindow(hwnd)
    {
        auto d3dDevice  = context.GetD3DDevice();
        auto dxgiDevice = d3dDevice.query<IDXGIDevice>();

        wil::com_ptr<IDXGIAdapter> dxgiAdapter;
        THROW_IF_FAILED(dxgiDevice->GetAdapter(dxgiAdapter.put()));

        wil::com_ptr<IDXGIFactory2> dxgiFactory;
        THROW_IF_FAILED(dxgiAdapter->GetParent(IID_PPV_ARGS(dxgiFactory.put())));

        DXGI_SWAP_CHAIN_DESC1 description{};
        description.Format           = DXGI_FORMAT_B8G8R8A8_UNORM;
        description.SampleDesc.Count = 1;
        description.BufferUsage      = DXGI_USAGE_RENDER_TARGET_OUTPUT;
        description.BufferCount      = 2;
        description.SwapEffect       = DXGI_SWAP_EFFECT_FLIP_SEQUENTIAL;

        THROW_IF_FAILED(dxgiFactory->CreateSwapChainForHwnd(d3dDevice.get(), m_TargetWindow, &description, nullptr, nullptr, m_SwapChain.put()));

        wil::com_ptr<IDXGISurface> backBuffer;
        THROW_IF_FAILED(m_SwapChain->GetBuffer(0, IID_PPV_ARGS(backBuffer.put())));

        D2D1_BITMAP_PROPERTIES1 bitmapProperties = D2D1::BitmapProperties1(
            D2D1_BITMAP_OPTIONS_TARGET | D2D1_BITMAP_OPTIONS_CANNOT_DRAW,
            D2D1::PixelFormat(DXGI_FORMAT_B8G8R8A8_UNORM, D2D1_ALPHA_MODE_PREMULTIPLIED),
            96.0f, // dpiX: 96縺ｫ蝗ｺ螳・
            96.0f  // dpiY: 96縺ｫ蝗ｺ螳・
        );

        THROW_IF_FAILED(context.GetD2DContext()->CreateBitmapFromDxgiSurface(backBuffer.get(), &bitmapProperties, m_TargetBitmap.put()));

        // 繝・ヰ繧､繧ｹ繧ｳ繝ｳ繝・く繧ｹ繝郁・菴薙↓繧・DPI 繧偵そ繝・ヨ縺吶ｋ
        context.GetD2DContext()->SetDpi(96.0f, 96.0f);
    }

    auto RenderTarget::Resize(const Device::Context& context, UINT width, UINT height) -> void
    {
        if (!m_SwapChain)
        {
            return;
        }

        // 蜿､縺・ち繝ｼ繧ｲ繝・ヨ繧堤ｴ譽・
        m_TargetBitmap.reset();

        // SwapChain 縺ｮ繧ｵ繧､繧ｺ螟画峩
        THROW_IF_FAILED(m_SwapChain->ResizeBuffers(
            0, // 繝舌ャ繝輔ぃ謨ｰ縺ｯ螟画峩縺励↑縺・
            width,
            height,
            DXGI_FORMAT_UNKNOWN,
            0
        ));

        // 譁ｰ縺励＞ BackBuffer 繧貞叙蠕・
        wil::com_ptr<IDXGISurface> backBuffer;
        THROW_IF_FAILED(m_SwapChain->GetBuffer(0, IID_PPV_ARGS(backBuffer.put())));

        // D2D Bitmap 蜀堺ｽ懈・
        D2D1_BITMAP_PROPERTIES1 bitmapProperties = D2D1::BitmapProperties1(
            D2D1_BITMAP_OPTIONS_TARGET | D2D1_BITMAP_OPTIONS_CANNOT_DRAW,
            D2D1::PixelFormat(DXGI_FORMAT_B8G8R8A8_UNORM, D2D1_ALPHA_MODE_PREMULTIPLIED),
            96.0f, // dpiX: 96縺ｫ蝗ｺ螳・
            96.0f  // dpiY: 96縺ｫ蝗ｺ螳・
        );

        THROW_IF_FAILED(context.GetD2DContext()->CreateBitmapFromDxgiSurface(backBuffer.get(), &bitmapProperties, m_TargetBitmap.put()));

        // 繝・ヰ繧､繧ｹ繧ｳ繝ｳ繝・く繧ｹ繝郁・菴薙↓繧・DPI 繧偵そ繝・ヨ縺吶ｋ
        context.GetD2DContext()->SetDpi(96.0f, 96.0f);
    }

    auto RenderTarget::Present() const noexcept -> HRESULT
    {
        return m_SwapChain->Present(1, 0);
    }

    auto Viewport::GetRenderTargetWindow() const noexcept -> HWND
    {
        wil::unique_event_nothrow createWindowCompletion;

        if (!createWindowCompletion.try_open(L"Local\\N503.CppWin32.Event.Core.CreateWindowCompletion", SYNCHRONIZE | EVENT_MODIFY_STATE))
        {
            return {};
        }

        if (::WaitForSingleObject(createWindowCompletion.get(), 0) != WAIT_OBJECT_0)
        {
            return {};
        }

        struct Context
        {
            HWND Result{};
            DWORD TargetPID{};
        };

        Context context{ nullptr, ::GetCurrentProcessId() };

        auto callback = [](HWND handle, LPARAM lParam) -> BOOL
        {
            auto pContext = reinterpret_cast<Context*>(lParam);

            DWORD pid{};
            ::GetWindowThreadProcessId(handle, &pid);

            if (pid == pContext->TargetPID)
            {
                wchar_t className[256]{};
                ::GetClassNameW(handle, className, 256);

                if (std::wcscmp(className, L"N503.CppWin32.Core.Window") == 0)
                {
                    pContext->Result = handle;
                    return FALSE;
                }
            }

            return TRUE;
        };

        ::EnumWindows(callback, reinterpret_cast<LPARAM>(&context));

        if (!context.Result)
        {
            return {};
        }
        else
        {
            // 謌仙粥縺励◆縺ｮ縺ｧ譌励ｒ荳九ｍ縺・
            createWindowCompletion.ResetEvent();
        }

        return context.Result;
    }

    auto CreateSprite::operator()(Message::Context& context) const -> void
    {
        auto handle = context.ResourceContainer.Add(Path);

        if (!handle)
        {
            return;
        }

        auto resource = context.ResourceContainer.Get(handle);

        if (!resource)
        {
            return;
        }

        try
        {
            auto entity = context.Registry.CreateEntity();

            auto& transform    = context.Registry.AddComponent(entity, System::Transform{});
            transform.X        = 0.0f;
            transform.Y        = 0.0f;
            transform.ScaleX   = 1.0f;
            transform.ScaleY   = 1.0f;
            transform.Rotation = 0.0f;

            auto& sprite  = context.Registry.AddComponent(entity, System::Sprite{ handle });
            sprite.Width  = static_cast<float>(resource->Pixels.Width);
            sprite.Height = static_cast<float>(resource->Pixels.Height);

            *Result = entity;
        }
        catch (...)
        {
            context.ResourceContainer.Remove(handle);
        }
    }

    auto Dispatcher::Dispatch(Queue& queue, Context& context) -> void
    {
        auto bundles = queue.PopAll();

        while (!bundles.empty())
        {
            auto& envelope = bundles.front();

            for (auto& packet : envelope.Packets)
            {
                std::visit([&context](auto&& concrete) { concrete(context); }, packet);
            }

            if (envelope.Signal)
            {
                envelope.Signal->release();
            }

            bundles.pop();
        }
    }

} // namespace N503::Renderer2D::Message

    struct Envelope
    {

        std::binary_semaphore* Signal{ nullptr };

        Envelope() = default;

        }

        Envelope(Packet&& packet, std::binary_semaphore* signal) : Signal(signal)

            Packets.reserve(1);

        }

        Envelope(std::vector<Packet>&& packets, std::binary_semaphore* signal) : Packets(std::move(packets)), Signal(signal)

        }

    auto Queue::Enqueue(Packet&& packet) -> void
    {
        bool isBusy = false;

#ifdef _DEBUG
        {
            const std::size_t currentSize = m_Buffer[m_BufferIndex].Container.size();
            const std::size_t capacity    = MaxMessageQueued;

            if (currentSize >= capacity * 0.8) // 80%繧定ｶ・∴縺溘ｉ隴ｦ蜻・
            {
                Engine::GetInstance().GetDiagnosticsSink().AddEntry(Diagnostics::Entry{
                    .Severity = Diagnostics::Severity::Warning,
                    .Expected = std::format("EventQueue is congesting: CurrentSize={}, Capacity={}\n", currentSize, capacity).data(),
                    .Position = 0,
                });
            }
        }
#endif
        {
            while (m_SpinLock.test_and_set(std::memory_order_acquire))
            {
                _mm_pause();
            }

            const std::size_t currentSize = m_Buffer[m_BufferIndex].Container.size();
            const std::size_t capacity    = MaxMessageQueued;

            if (currentSize >= capacity * 0.95)
            {
                isBusy = true;
            }
            else
            {
                m_Buffer[m_BufferIndex].Container.push(Envelope{ std::move(packet) });
            }

            m_SpinLock.clear(std::memory_order_release);
        }

        if (isBusy)
        {
            EnqueueSync(std::move(packet));
        }
        else
        {
            m_WakeupEvent.SetEvent();
        }
    }

    auto Queue::EnqueueSync(Packet&& packet) -> void
    {
        std::binary_semaphore signal{ 0 };

#ifdef _DEBUG
        {
            const std::size_t currentSize = m_Buffer[m_BufferIndex].Container.size();
            const std::size_t capacity    = MaxMessageQueued;

            if (currentSize >= capacity * 0.8) // 80%繧定ｶ・∴縺溘ｉ隴ｦ蜻・
            {
                Engine::GetInstance().GetDiagnosticsSink().AddEntry(Diagnostics::Entry{
                    .Severity = Diagnostics::Severity::Warning,
                    .Expected = std::format("EventQueue is congesting: CurrentSize={}, Capacity={}\n", currentSize, capacity).data(),
                    .Position = 0 });
            }
        }
#endif
        {
            while (m_SpinLock.test_and_set(std::memory_order_acquire))
            {
                _mm_pause();
            }

            m_Buffer[m_BufferIndex].Container.push(Envelope{ std::move(packet), &signal });

            m_SpinLock.clear(std::memory_order_release);
        }

        m_WakeupEvent.SetEvent();

        signal.acquire();
    }

    auto Queue::Enqueue(std::vector<Packet>&& packets) -> void
    {
        while (!packets.empty())
        {
            while (m_SpinLock.test_and_set(std::memory_order_acquire))
            {
                _mm_pause();
            }

            const std::size_t currentSize  = m_Buffer[m_BufferIndex].Container.size();
            const std::size_t canPushCount = (currentSize < MaxMessageQueued) ? (MaxMessageQueued - currentSize) : 0;
            const std::size_t processCount = std::min(packets.size(), canPushCount);

            if (processCount > 0)
            {
                auto first = packets.begin();
                auto last  = packets.begin() + processCount;

                std::vector<Packet> bundle(
                    std::make_move_iterator(first),
                    std::make_move_iterator(last)
                );

                packets.erase(first, last);

                m_Buffer[m_BufferIndex].Container.push(Envelope{ std::move(bundle), nullptr });
                m_WakeupEvent.SetEvent();
            }

            m_SpinLock.clear(std::memory_order_release);

            if (!packets.empty())
            {
                std::this_thread::sleep_for(std::chrono::milliseconds(1));
            }
        }
    }

    auto Queue::EnqueueSync(std::vector<Packet>&& packets) -> void
    {
        std::binary_semaphore signal{ 0 };

        while (!packets.empty())
        {
            while (m_SpinLock.test_and_set(std::memory_order_acquire))
            {
                _mm_pause();
            }

            const std::size_t currentSize  = m_Buffer[m_BufferIndex].Container.size();
            const std::size_t canPushCount = (currentSize < MaxMessageQueued) ? (MaxMessageQueued - currentSize) : 0;
            const std::size_t processCount = std::min(packets.size(), canPushCount);

            if (processCount > 0)
            {
                auto first = packets.begin();
                auto last  = packets.begin() + processCount;

                std::vector<Packet> bundle(
                    std::make_move_iterator(first),
                    std::make_move_iterator(last)
                );

                packets.erase(first, last);

                m_Buffer[m_BufferIndex].Container.push(Envelope{ std::move(bundle), &signal });
                m_WakeupEvent.SetEvent();
            }

            m_SpinLock.clear(std::memory_order_release);

            if (!packets.empty())
            {
                std::this_thread::sleep_for(std::chrono::milliseconds(1));
            }
        }

        signal.acquire();
    }

    /// @brief
    /// @return
    [[nodiscard]]
    auto Queue::PopAll() -> Container
    {
        while (m_SpinLock.test_and_set(std::memory_order_acquire))
        {
            _mm_pause();
        }

        // UI繧ｹ繝ｬ繝・ラ縺ｧ貅懊ａ縺溘ョ繝ｼ繧ｿ繧偵窟llocator縺斐→縲榊･ｪ縺・悉繧・
        auto container = std::move(m_Buffer[m_BufferIndex].Container);

        // 遨ｺ縺ｫ縺ｪ縺｣縺溷ｴ謇縺ｫ縲∵ｭ｣縺励＞ Allocator・・torage縺ｸ縺ｮ繝昴う繝ｳ繧ｿ・峨ｒ蜀崎｣・｡ｫ
        m_Buffer[m_BufferIndex].Container = Container{ m_Buffer[m_BufferIndex].Allocator };

        // 谺｡縺ｮ Push 逕ｨ縺ｫ繧､繝ｳ繝・ャ繧ｯ繧ｹ繧貞・繧頑崛縺医ｋ
        m_BufferIndex = (m_BufferIndex + 1) % m_Buffer.size();

        m_SpinLock.clear(std::memory_order_release);

        return container;
    }

    /// @brief
    /// @return
    [[nodiscard]]
    auto Queue::GetWakeupEventHandle() const -> HANDLE
    {
        return m_WakeupEvent.get();
    }

namespace N503::Renderer2D::Message

    class Queue final
    {
        static const std::size_t MaxMessageQueued = 4096;

        static const std::size_t BufferCount = 3;

    public:
        using Element = Envelope;

        struct Slot
        {
            alignas(void*) std::byte data[sizeof(Element) + 32];
        };

        using Storage = ::N503::Memory::Storage::Pool<Slot>;

        using Allocator = ::N503::Memory::StorageAllocator<Element, Storage>;

        using Strategy = std::list<Element, Allocator>;

        using Container = std::queue<Element, Strategy>;

    public:
        template <typename TPacket> class MessageBorrower
        {
        public:
            MessageBorrower(std::unique_lock<std::mutex>&& lock, Container& queue) : m_Lock(std::move(lock))
            {
                if (!queue.empty() && std::holds_alternative<TPacket>(queue.back().Packet))
                {
                    m_Target = &std::get_if<TPacket>(queue.back().Packet);
                }
            }

            MessageBorrower(const MessageBorrower&) = delete;

            auto operator=(const MessageBorrower&) -> MessageBorrower& = delete;

            auto operator->() -> TPacket*
            {
                return m_Target;
            }

            explicit operator bool() const
            {
                return m_Target != nullptr;
            }

        private:
            std::unique_lock<std::mutex> m_Lock;

            TPacket* m_Target = nullptr;
        };

    public:
        Queue() = default;

        Queue(const Queue&) = delete;

        auto operator=(const Queue&) -> Queue& = delete;

        Queue(Queue&&) = delete;

        auto operator=(Queue&&) -> Queue& = delete;

    public:
        auto Enqueue(Packet&&) -> void;

        auto EnqueueSync(Packet&&) -> void;

        auto Enqueue(std::vector<Packet>&&) -> void;

        auto EnqueueSync(std::vector<Packet>&&) -> void;

        [[nodiscard]]
        auto PopAll() -> Container;

        [[nodiscard]]
        auto GetWakeupEventHandle() const -> HANDLE;

    public:
        template <typename TDataType> auto TryBorrowBack() -> MessageBorrower<TDataType>
        {
            // 繝ｭ繝・け繧貞･ｪ蜿悶☆繧・
            std::unique_lock<std::mutex> lock{ m_Mutex };

            // 繝ｭ繝・け縺励◆迥ｶ諷九〒 Container 繧堤音螳壹＠縲〕ock 縺斐→ Borrower 縺ｫ貂｡縺・
            return MessageBorrower<TDataType>{ std::move(lock), m_Buffer[m_BufferIndex].Container };
        }

    private:
        /// @brief
        struct Buffer
        {
            typename Queue::Storage Storage{ MaxMessageQueued }; ///< Allocator 繧医ｊ蜈医↓螳｣險縺吶ｋ蠢・ｦ√′縺ゅｋ莠九↓豕ｨ諢・

            typename Queue::Allocator Allocator{ &Storage }; ///< Container 繧医ｊ蜈医↓螳｣險縺吶ｋ蠢・ｦ√′縺ゅｋ莠九↓豕ｨ諢・

            typename Queue::Container Container{ Allocator };
        };

        std::array<Buffer, BufferCount> m_Buffer{};

        std::uint32_t m_BufferIndex{ 0 };

        wil::unique_event_nothrow m_WakeupEvent{ ::CreateEventW(nullptr, FALSE, FALSE, nullptr) };

        std::mutex m_Mutex;

        std::atomic_flag m_SpinLock = ATOMIC_FLAG_INIT;
    };

    auto Container::Add(const std::string_view path) -> ResourceHandle
    {
        // 縺吶〒縺ｫ蜷後§繝代せ縺ｮ繝ｪ繧ｽ繝ｼ繧ｹ縺悟ｭ伜惠縺吶ｋ蝣ｴ蜷医・縲√◎縺ｮ繝上Φ繝峨Ν繧定ｿ斐＠縺ｾ縺吶・
        if (auto it = m_Indexes.find(path); it != m_Indexes.end())
        {
            return it->second;
        }

        // 蛻ｩ逕ｨ蜿ｯ閭ｽ縺ｪ繝上Φ繝峨Ν縺後↑縺・ｴ蜷医・縲∫┌蜉ｹ縺ｪ繝上Φ繝峨Ν繧定ｿ斐＠縺ｾ縺吶・
        if (m_AvailableHandles.empty())
        {
            return { .ID = Handle::ResourceID::Invalid };
        }

        // 蛻ｩ逕ｨ蜿ｯ閭ｽ縺ｪ繝上Φ繝峨Ν縺ｮ繝ｪ繧ｹ繝医°繧峨ワ繝ｳ繝峨Ν繧貞叙蠕励＠縺ｾ縺吶・
        ResourceHandle handle = m_AvailableHandles.back();

        // 繝上Φ繝峨Ν縺ｮID繧偵う繝ｳ繝・ャ繧ｯ繧ｹ縺ｨ縺励※菴ｿ逕ｨ縺励∪縺吶ゅ％繧後↓繧医ｊ縲＾(1)縺ｧ繧ｨ繝ｳ繝医Μ縺ｫ繧｢繧ｯ繧ｻ繧ｹ縺ｧ縺阪∪縺吶・
        const auto index = static_cast<std::uint64_t>(handle.ID);

        // 逕ｻ蜒上ｒ繝・さ繝ｼ繝峨＠縺ｦ縲∫｢ｺ菫昴＠縺溘Γ繝｢繝ｪ縺ｫ繝斐け繧ｻ繝ｫ繝・・繧ｿ繧呈嶌縺崎ｾｼ縺ｿ縺ｾ縺吶・
        Codec::WicImageDecoder decoder(path);

        Pixels::Buffer pixels;
        pixels.Size = static_cast<std::size_t>(decoder.GetHeight()) * static_cast<std::size_t>(decoder.GetPitch());

        // 逕ｻ蜒上ョ繝ｼ繧ｿ縺ｮ繝舌う繝医し繧､繧ｺ縺ｫ蝓ｺ縺･縺・※縲√い繝ｪ繝｡繝ｳ繝医ｒ16繝舌う繝医→縺励◆逕溘・繝舌う繝亥・縺ｨ縺励※繝｡繝｢繝ｪ繧堤｢ｺ菫昴＠縺ｾ縺吶・
        void* address = m_Storage.AllocateBytes(pixels.Size, 16);

        if (address == nullptr)
        {
            return {}; // 繝｡繝｢繝ｪ遒ｺ菫昴↓螟ｱ謨励＠縺溷ｴ蜷医・縲∫┌蜉ｹ縺ｪ繝上Φ繝峨Ν繧定ｿ斐＠縺ｾ縺吶・
        }

        pixels = decoder.Decode(
            [&](std::size_t size) -> std::span<std::byte>
            {
                if (size > pixels.Size)
                {
                    return {};
                }
                return std::span<std::byte>(static_cast<std::byte*>(address), size);
            }
        );

        // 繧ｨ繝ｳ繝医Μ繧剃ｽ懈・縺励※菫晏ｭ倥＠縺ｾ縺吶・
        m_Entries[index] = Entry{
            .Handle   = handle,
            .Pixels   = pixels,
            .Metadata = { .Path = std::string(path) },
        };

        // 繝代せ縺ｨ繝上Φ繝峨Ν縺ｮ蟇ｾ蠢懊ｒ菫晏ｭ倥＠縺ｾ縺吶・
        m_Indexes[m_Entries[index].Metadata.Path] = handle;

#ifdef _DEBUG
        auto log = std::format("[Renderer2D] <Resource::Container>: Add to ResourceID={}", static_cast<std::uint64_t>(handle.ID));
        Engine::GetInstance().GetDiagnosticsSink().AddEntry(Diagnostics::Entry{ Diagnostics::Severity::Verbose, log });
#endif

        // 蛻ｩ逕ｨ蜿ｯ閭ｽ縺ｪ繝上Φ繝峨Ν縺ｮ繝ｪ繧ｹ繝医°繧峨ワ繝ｳ繝峨Ν繧貞炎髯､縺励∪縺吶・
        m_AvailableHandles.pop_back();

        return handle;
    }

    auto Container::Remove(const ResourceHandle handle) -> bool
    {
        // 繝上Φ繝峨Ν縺ｮID繧偵う繝ｳ繝・ャ繧ｯ繧ｹ縺ｨ縺励※菴ｿ逕ｨ縺励∪縺吶・
        const auto index = static_cast<std::uint64_t>(handle.ID);

        if (index >= MaxEntries)
        {
            return false;
        }

        // 繝上Φ繝峨Ν縺ｮ荳紋ｻ｣縺後お繝ｳ繝医Μ縺ｮ荳紋ｻ｣縺ｨ荳閾ｴ縺励↑縺・ｴ蜷医・縲√☆縺ｧ縺ｫ辟｡蜉ｹ縺ｧ縺ゅｋ縺九√≠繧九＞縺ｯ荳紋ｻ｣縺悟商縺・→縺ｿ縺ｪ縺励∪縺吶・
        if (m_Entries[index].Handle.Generation != handle.Generation)
        {
            return false; // 縺吶〒縺ｫ辟｡蜉ｹ縲√≠繧九＞縺ｯ荳紋ｻ｣縺悟商縺・
        }

        // 繝代せ縺ｨ繝上Φ繝峨Ν縺ｮ蟇ｾ蠢懊ｒ蜑企勁縺励∪縺吶・
        m_Indexes.erase(m_Entries[index].Metadata.Path);

        // 繧ｨ繝ｳ繝医Μ繧堤ｩｺ縺ｫ縺励※縲∵ｬ｡縺ｮ荳紋ｻ｣縺ｮ繝上Φ繝峨Ν繧貞茜逕ｨ蜿ｯ閭ｽ縺ｪ繝上Φ繝峨Ν縺ｮ繝ｪ繧ｹ繝医↓霑ｽ蜉縺励∪縺吶・
        ResourceHandle nextGenerationHandle = handle;
        nextGenerationHandle.Generation     = static_cast<Handle::Generation>(static_cast<std::uint64_t>(handle.Generation) + 1);

        // 繧ｨ繝ｳ繝医Μ繧堤ｩｺ縺ｫ縺励∪縺吶・
        m_Entries[index] = {};

        // 谺｡縺ｮ荳紋ｻ｣縺ｮ繝上Φ繝峨Ν繧貞茜逕ｨ蜿ｯ閭ｽ縺ｪ繝上Φ繝峨Ν縺ｮ繝ｪ繧ｹ繝医↓霑ｽ蜉縺励∪縺吶・
        m_AvailableHandles.push_back(nextGenerationHandle);

        return true;
    }

    auto Container::Get(const ResourceHandle handle) const -> const Entry*
    {
        const auto index = static_cast<std::uint64_t>(handle.ID);

        if (index >= MaxEntries)
        {
            return nullptr;
        }

        if (m_Entries[index].Handle.Generation != handle.Generation)
        {
            return nullptr; // 辟｡蜉ｹ縲√≠繧九＞縺ｯ荳紋ｻ｣縺悟商縺・
        }

        return &m_Entries[index];
    }

    auto Container::Get(const std::string_view path) const -> const Entry*
    {
        if (auto it = m_Indexes.find(path); it != m_Indexes.end())
        {
            return Get(it->second);
        }

        return nullptr;
    }

    auto Container::GetHandle(const std::string_view path) const -> ResourceHandle
    {
        if (auto it = m_Indexes.find(path); it != m_Indexes.end())
        {
            return it->second;
        }

        return { .ID = Handle::ResourceID::Invalid };
    }

    Registry::Registry() : m_World(std::make_unique<World>())
    {
        for (std::uint64_t entityId = 0; entityId < MaxEntities; ++entityId)
        {
            m_AvailableEntityIds.push_back(static_cast<Entity>(MaxEntities - entityId));

    auto Registry::CreateEntity() -> Entity
    {
        if (m_AvailableEntityIds.empty())
        {
            throw std::runtime_error("Maximum number of entities reached.");
        }

        // 蛻ｩ逕ｨ蜿ｯ閭ｽ縺ｪ繧ｨ繝ｳ繝・ぅ繝・ぅID繧貞叙蠕・
        auto entityID = m_AvailableEntityIds.back();
        m_AvailableEntityIds.pop_back();

        // 繧ｨ繝ｳ繝・ぅ繝・ぅ繧呈怏蜉ｹ蛹・
        auto index = ToArrayIndex(entityID);
        m_World->AliveBits.set(index);
        m_World->ComponentMasks[index].reset();

        // 繧ｳ繝ｳ繝昴・繝阪Φ繝医ｒ蛻晄悄蛹・
        m_World->Transforms[index].Reset();
        m_World->Sprites[index].Reset();
        m_World->Texts[index].Reset();

        return entityID;
    }

    auto Registry::DestroyEntity(Entity entity) -> void
    {
        const auto index = ToArrayIndex(entity);

        if (ToArrayIndex(entity) >= MaxEntities)
        {
            throw std::runtime_error("Invalid entity ID.");
        }

        // 繧ｨ繝ｳ繝・ぅ繝・ぅ繧堤┌蜉ｹ蛹・
        m_World->AliveBits.reset(index);
        m_World->ComponentMasks[index].reset();

        // 繧ｳ繝ｳ繝昴・繝阪Φ繝医ｒ繝ｪ繧ｻ繝・ヨ
        m_World->Transforms[index].Reset();
        m_World->Sprites[index].Reset();
        m_World->Texts[index].Reset();

        // 繧ｨ繝ｳ繝・ぅ繝・ぅID繧貞・蛻ｩ逕ｨ蜿ｯ閭ｽ縺ｪ繝ｪ繧ｹ繝医↓謌ｻ縺・
        m_AvailableEntityIds.push_back(entity);
    }

    template <typename T> struct ComponentTraits;

    template <> struct ComponentTraits<Transform>
    {
        static constexpr ComponentType Type = ComponentType::Transform;
        static auto& Get(World& world, std::size_t index)
        {
            return world.Transforms[index];
        }

        static constexpr ComponentType Type = ComponentType::Sprite;
        static auto& Get(World& world, std::size_t index)
        {
            return world.Sprites[index];
        }
    };

    template <> struct ComponentTraits<Text>
    {
        static constexpr ComponentType Type = ComponentType::Text;
        static auto& Get(World& world, std::size_t index)
        {
            return world.Texts[index];
        }
    };

    // -----------------------------------
    // View
    // -----------------------------------
    template <typename... Components> class View
    {
        World& m_World;

        std::bitset<MaxComponents> m_Mask;

    public:
        View(World& world) : m_World(world)
        {
            ((m_Mask.set(static_cast<std::size_t>(GetComponentType<Components>()))), ...);
        }

        struct Iterator
        {
            World& World;
            std::bitset<MaxComponents> Mask;
            std::size_t Index;

            auto operator*() const -> Entity
            {
                return static_cast<Entity>(Index);
            }

            auto operator++() -> Iterator&
            {
                do
                {
                    Index++;
                } while (Index < MaxEntities && (!World.AliveBits.test(Index) || (World.ComponentMasks[Index] & Mask) != Mask));
                return *this;
            }

            auto operator!=(const Iterator& other) const -> bool
            {
                return Index != other.Index;
            }
        };

        auto begin()
        {
            std::size_t first = 0;
            while (first < MaxEntities && (!m_World.AliveBits.test(first) || (m_World.ComponentMasks[first] & m_Mask) != m_Mask))
            {
                first++;
            }
            return Iterator{ m_World, m_Mask, first };
        }

        auto end()
        {
            return Iterator{ m_World, m_Mask, MaxEntities };
        }

    private:
        template <typename T> static constexpr auto GetComponentType() -> ComponentType
        {
            return ComponentTraits<T>::Type;
        }
    };

    // -----------------------------------
    // Registry
    // -----------------------------------
    class Registry
    {
    public:
        Registry();

        auto CreateEntity() -> Entity;

        auto DestroyEntity(Entity entity) -> void;

        template <typename T> auto AddComponent(Entity entity, T&& component) -> T&
        {
            const auto index = ToArrayIndex(entity);

            auto& entry = ComponentTraits<T>::Get(*m_World, index);
            entry       = std::forward<T>(component);

            m_World->ComponentMasks[index].set(static_cast<size_t>(ComponentTraits<T>::Type));
            return entry;
        }

        template <typename T> auto GetComponent(Entity entity) -> T&
        {
            return ComponentTraits<T>::Get(*m_World, ToArrayIndex(entity));
        }

        template <typename T> bool HasComponent(Entity entity) const
        {
            return m_World->ComponentMasks[ToArrayIndex(entity)].test(static_cast<size_t>(ComponentTraits<T>::Type));
        }

        template <typename T> void RemoveComponent(Entity entity)
        {
            m_World->ComponentMasks[ToArrayIndex(entity)].reset(static_cast<size_t>(ComponentTraits<T>::Type));
        }

        template <typename... Components> [[nodiscard]] auto GetView() -> View<Components...>
        {
            return View<Components...>(*m_World);
        }

    private:
        std::vector<Entity> m_AvailableEntityIds;

        std::unique_ptr<Renderer2D::System::World> m_World;
    };

    auto RendererSystem::Update(Registry& registry, Device::Context& context) -> void
    {
        // 謠冗判蟇ｾ雎｡縺ｮ繧ｨ繝ｳ繝・ぅ繝・ぅ繧貞庶髮・
        std::vector<Entity> renderList;
        for (auto entity : registry.GetView<Transform, Sprite>())
        {
            renderList.push_back(entity);
        }

        // Z蛟､縺ｧ繧ｽ繝ｼ繝茨ｼ亥ｰ上＆縺・・ｼ晏･･縲∝､ｧ縺阪＞鬆・ｼ晄焔蜑搾ｼ・
        std::sort(
            renderList.begin(),
            renderList.end(),
            [&](Entity a, Entity b) { return registry.GetComponent<Transform>(a).Z < registry.GetComponent<Transform>(b).Z; }
        );

        for (auto entity : renderList)
        {
            auto& transform = registry.GetComponent<Transform>(entity);
            auto& sprite    = registry.GetComponent<Sprite>(entity);

            if (sprite.Bitmap)
            {
                // 1. 陦悟・繧剃ｽ懈・・・S縺ｮD2D1蜷榊燕遨ｺ髢薙・繝倥Ν繝代・繧剃ｽｿ縺・→讌ｽ縺ｧ縺呻ｼ・
                // 諡｡螟ｧ -> 蝗櫁ｻ｢ -> 遘ｻ蜍・縺ｮ鬆・ｺ上ｒ豁ｻ螳茨ｼ・
                // 荳ｭ蠢・せ繧定ｨ育ｮ暦ｼ井ｾ具ｼ夂判蜒上・逵溘ｓ荳ｭ・・
                float centerX = sprite.Width / 2.0f;
                float centerY = sprite.Height / 2.0f;

                // 1. 蜴溽せ繧剃ｸｭ蠢・↓謖√▲縺ｦ縺上ｋ (-centerX, -centerY)
                // 2. 諡｡螟ｧ繝ｻ蝗櫁ｻ｢
                // 3. 逕ｻ髱｢荳翫・逶ｮ逧・慍縺ｸ遘ｻ蜍・(transform.X, transform.Y)
                auto matrix = D2D1::Matrix3x2F::Translation(-centerX, -centerY) * D2D1::Matrix3x2F::Scale(transform.ScaleX, transform.ScaleY) *
                              D2D1::Matrix3x2F::Rotation(transform.Rotation) * D2D1::Matrix3x2F::Translation(transform.X, transform.Y);

                // 2. 陦悟・繧帝←逕ｨ
                context.SetTransform(matrix);

                // 3. 謠冗判遏ｩ蠖｢・医Ο繝ｼ繧ｫ繝ｫ蠎ｧ讓呻ｼ・
                // 縺薙％縺ｫ transform.X 繧貞・繧後※縺ｯ縺・￠縺ｾ縺帙ｓ縲・
                // (0, 0) 繧貞渕貅悶→縺励◆縲檎ｵｵ縺昴・繧ゅ・縺ｮ繧ｵ繧､繧ｺ縲阪ｒ謖・ｮ壹＠縺ｾ縺吶・
                D2D1_RECT_F destRect = { 0, 0, sprite.Width, sprite.Height };

                context.DrawBitmap(sprite.Bitmap, destRect);
            }
        }

        // 谺｡縺ｮ謠冗判縺ｫ蠖ｱ髻ｿ縺励↑縺・ｈ縺・∬｡悟・繧偵Μ繧ｻ繝・ヨ・亥腰菴崎｡悟・・・
        context.SetTransform(D2D1::Matrix3x2F::Identity());
    }

    auto SpriteSystem::Update(Registry& registry, Device::Context& context, Resource::Container& container) -> void
    {
        for (auto entity : registry.GetView<Sprite>())
        {
            auto& sprite = registry.GetComponent<Sprite>(entity);

            if (!sprite.ResourceHandle || sprite.Bitmap)
            {
                continue;
            }

            // GPU蛛ｴ縺ｫ繝ｪ繧ｽ繝ｼ繧ｹ繧剃ｽ懈・貂医∩縺ｮ蝣ｴ蜷医・蜀榊茜逕ｨ縺吶ｋ
            sprite.Bitmap = context.GetResourceCache().Get(sprite.ResourceHandle);

            if (sprite.Bitmap)
            {
                continue;
            }

            // GPU蛛ｴ縺ｫ繝ｪ繧ｽ繝ｼ繧ｹ縺後↑縺代ｌ縺ｰ菴懈・縺吶ｋ
            auto resource = container.Get(sprite.ResourceHandle);

            if (!resource)
            {
                continue;
            }

            sprite.Bitmap = context.GetResourceCache().GetOrCreateBitmap(sprite.ResourceHandle, resource->Pixels);

            if (!sprite.Bitmap)
            {
                // 繝薙ャ繝医・繝・・縺ｮ菴懈・縺ｫ螟ｱ謨励＠縺溷ｴ蜷医・縲√Μ繧ｽ繝ｼ繧ｹ繝上Φ繝峨Ν繧堤┌蜉ｹ蛹悶☆繧・
                sprite.ResourceHandle = { .ID = Handle::ResourceID::Invalid, .Type = Handle::ResourceType::None, .Generation = Handle::Generation::Default };
            }
        }
    }

        // 縺吶〒縺ｫ繧ｨ繝ｳ繧ｸ繝ｳ縺碁幕蟋九＆繧後※縺・ｋ蝣ｴ蜷医・縲∽ｽ輔ｂ縺励↑縺・
        if (m_IsRunning.load(std::memory_order_acquire))
        {
            return;
        }

        std::binary_semaphore signal{ 0 };

        // 繝ｬ繝ｳ繝繝ｩ繝ｼ 繧ｹ繝ｬ繝・ラ繧剃ｽ懈・縺吶ｋ
        m_RendererThread = std::jthread(
            [this, &signal](const std::stop_token stopToken)
            {
                // 繧ｹ繝ｬ繝・ラ蜷阪ｒ險ｭ螳壹☆繧・
                ::SetThreadDescription(::GetCurrentThread(), L"N503.CppWin32.Renderer2D");

                // 繝ｬ繝ｳ繝繝ｩ繝ｼ 繧ｹ繝ｬ繝・ラ縺ｮ蛻晄悄蛹悶′螳御ｺ・＠縺溘％縺ｨ繧帝夂衍縺吶ｋ
                signal.release();

                // 繝ｬ繝ｳ繝繝ｩ繝ｼ 繧ｹ繝ｬ繝・ラ縺ｮ蛻晄悄蛹悶′螳御ｺ・＠縺溘％縺ｨ繧堤､ｺ縺吶う繝吶Φ繝医ｒ繧ｷ繧ｰ繝翫Ν迥ｶ諷九↓縺吶ｋ
                // m_StartedEvent.SetEvent();

                // COM 繧貞・譛溷喧縺吶ｋ
                auto coinit = wil::CoInitializeEx(COINIT_MULTITHREADED);

                // PeekMessageW繧貞他縺ｳ蜃ｺ縺励※縲√Ξ繝ｳ繝繝ｩ繝ｼ繧ｹ繝ｬ繝・ラ縺ｮ繝｡繝・そ繝ｼ繧ｸ繧ｭ繝･繝ｼ繧剃ｽ懈・縺吶ｋ
                MSG msg{};
                ::PeekMessageW(&msg, nullptr, WM_USER, WM_USER, PM_NOREMOVE);

                // 繝ｬ繝ｳ繝繝ｩ繝ｼ繧ｹ繝ｬ繝・ラ繧帝幕蟋九☆繧・
                Run(stopToken);

                // 繝ｬ繝ｳ繝繝ｩ繝ｼ 繧ｹ繝ｬ繝・ラ縺悟●豁｢縺励◆縺薙→繧堤､ｺ縺吶う繝吶Φ繝医ｒ繝ｪ繧ｻ繝・ヨ迥ｶ諷九↓縺吶ｋ
                m_IsRunning.store(false, std::memory_order_release);
            }
        );

        // 繧ｹ繝ｬ繝・ラ縺瑚ｵｷ蜍輔＠縺ｦ蛻晄悄蛹悶′螳御ｺ・☆繧九∪縺ｧ蠕・ｩ溘☆繧・
        signal.acquire();

        // 繝ｬ繝ｳ繝繝ｩ繝ｼ繧ｹ繝ｬ繝・ラ縺梧緒逕ｻ蜿ｯ閭ｽ縺ｪ迥ｶ諷九↓縺ｪ繧九∪縺ｧ蠕・ｩ溘☆繧・
        m_StartedEvent.wait();

        // 繧ｨ繝ｳ繧ｸ繝ｳ縺碁幕蟋九＆繧後◆縺薙→繧堤､ｺ縺吶う繝吶Φ繝医ｒ繧ｷ繧ｰ繝翫Ν迥ｶ諷九↓縺吶ｋ
        m_IsRunning.store(true, std::memory_order_release);
    }

    auto Engine::Stop() -> void
    {
        if (!::PostThreadMessage(::GetThreadId(m_RendererThread.native_handle()), WM_QUIT, 0, 0))
        {
            const Diagnostics::Entry entry{
                .Severity = Diagnostics::Severity::Error,
                .Expected = std::format("PostThreadMessage failed: Reason={}, Handle={}\n", ::GetLastError(), m_RendererThread.native_handle()).data(),
                .Position = 0
            };

            m_DiagnosticsSink.AddEntry(entry);
        }
    }

    auto Engine::Wait() -> void
    {
        if (m_RendererThread.joinable())
        {
            m_RendererThread.join();
        }
    }

    auto Engine::Run(const std::stop_token stopToken) -> void
    {
        Diagnostics::Reporter diagnosticsReporter;
        diagnosticsReporter.AddSink(std::make_shared<Diagnostics::ConsoleSink>());

        auto resources     = std::make_unique<Resource::Container>();
        auto deviceContext = std::make_unique<Device::Context>();

        auto spriteSystem   = std::make_unique<System::SpriteSystem>();
        auto textSystem     = std::make_unique<System::TextSystem>();
        auto rendererSystem = std::make_unique<System::RendererSystem>();

        std::unique_ptr<Device::RenderTarget> renderTarget = nullptr;

        // 繝｡繝・そ繝ｼ繧ｸ 繝・ぅ繧ｹ繝代ャ繝√Ε繝ｼ繧貞・譛溷喧縺吶ｋ
        Message::Dispatcher messageDispatcher;

        // 繝ｬ繝ｳ繝繝ｩ繝ｼ繧ｹ繝ｬ繝・ラ縺ｮ邨ゆｺ・凾縺ｫ繝ｪ繧ｽ繝ｼ繧ｹ繧偵け繝ｪ繝ｼ繝ｳ繧｢繝・・縺吶ｋ縺溘ａ縺ｮ繧ｹ繧ｳ繝ｼ繝・繧ｬ繝ｼ繝・
        auto cleanupResources = wil::scope_exit(
            [&]
            {
                spriteSystem.reset();
                textSystem.reset();
                rendererSystem.reset();

                m_SystemRegistry.reset();

                renderTarget.reset();
                deviceContext.reset();
            }
        );

        // OS 繝｡繝・そ繝ｼ繧ｸ繧偵ョ繧｣繧ｹ繝代ャ繝√☆繧九Λ繝繝蠑・
        auto OSMessageDispatch = []() -> bool
        {
            MSG message{};

            while (::PeekMessage(&message, nullptr, 0, 0, PM_REMOVE))
            {
                if (message.message == WM_QUIT)
                {
                    return false;
                }

                ::TranslateMessage(&message);
                ::DispatchMessageW(&message);
            }

            return true;
        };

        // 繝ｬ繝ｳ繝繝ｩ繝ｼ 繧ｹ繝ｬ繝・ラ縺ｮ繝｡繧､繝ｳ 繝ｫ繝ｼ繝・
        auto waitHandles = { m_MessageQueue->GetWakeupEventHandle() };
        auto isAnyActive = true;

        while (!stopToken.stop_requested())
        {
            auto start = std::chrono::steady_clock::now();

            HWND hwnd = nullptr;

            if (renderTarget == nullptr && (hwnd = Device::Viewport::GetInstance().GetRenderTargetWindow()))
            {
                renderTarget = std::make_unique<Device::RenderTarget>(*deviceContext, hwnd);
                deviceContext->SetRenderTarget(renderTarget.get());
                m_StartedEvent.SetEvent();
            }

            // 繝｡繝・そ繝ｼ繧ｸ繧ｭ繝･繝ｼ縺ｮ繧ｦ繧ｧ繧､繧ｯ繧｢繝・・繧､繝吶Φ繝医→OS繝｡繝・そ繝ｼ繧ｸ繧貞ｾ・ｩ溘☆繧・
            const auto count   = static_cast<DWORD>(waitHandles.size());
            const auto handles = waitHandles.begin();
            const auto timeout = isAnyActive ? 0 : INFINITE;

            auto result = ::MsgWaitForMultipleObjectsEx(count, handles, timeout, QS_ALLINPUT, MWMO_INPUTAVAILABLE);

            if (result >= WAIT_OBJECT_0 && result < (WAIT_OBJECT_0 + count))
            {
                Message::Context context{
                    .ResourceContainer = *resources,
                    .Registry          = *m_SystemRegistry,
                    .DeviceContext     = *deviceContext,
                };
                messageDispatcher.Dispatch(*m_MessageQueue, context);
            }
            else if (result == WAIT_OBJECT_0 + count)
            {
                if (!OSMessageDispatch())
                {
                    return;
                }
            }

            // 繧ｳ繝槭Φ繝峨Μ繧ｹ繝医↓繧ｳ繝槭Φ繝峨′蟄伜惠縺吶ｋ蝣ｴ蜷医・縲√ョ繝舌う繧ｹ繧ｳ繝ｳ繝・く繧ｹ繝医ｒ菴ｿ逕ｨ縺励※繧ｳ繝槭Φ繝峨ｒ螳溯｡後＠縲√Ξ繝ｳ繝繝ｪ繝ｳ繧ｰ繧ｿ繝ｼ繧ｲ繝・ヨ繧呈署遉ｺ縺励∪縺吶・
            if (deviceContext->BeginDraw({ 0.1f, 0.2f, 0.4f, 1.0f }))
            {
                spriteSystem->Update(*m_SystemRegistry, *deviceContext, *resources);
                rendererSystem->Update(*m_SystemRegistry, *deviceContext);
                textSystem->Update(*m_SystemRegistry, *deviceContext);

                const auto endDrawResult = deviceContext->EndDraw();
                const auto presentResult = renderTarget->Present();

                if (endDrawResult == D2DERR_RECREATE_TARGET || presentResult == DXGI_ERROR_DEVICE_REMOVED || presentResult == DXGI_ERROR_DEVICE_RESET)
                {
                    // 繝・ヰ繧､繧ｹ縺悟炎髯､縺輔ｌ縺溘°繝ｪ繧ｻ繝・ヨ縺輔ｌ縺溷ｴ蜷医・縲√ョ繝舌う繧ｹ 繧ｳ繝ｳ繝・く繧ｹ繝医→繝ｬ繝ｳ繝繝ｪ繝ｳ繧ｰ 繧ｿ繝ｼ繧ｲ繝・ヨ繧貞・菴懈・縺吶ｋ
                    deviceContext = std::make_unique<Device::Context>();
                    renderTarget  = std::make_unique<Device::RenderTarget>(*deviceContext, renderTarget->GetTargetWindow());
                }
            }

            diagnosticsReporter.Submit(m_DiagnosticsSink);

            auto end = std::chrono::steady_clock::now();
            std::cout << "[Renderer2D] <Profile> : " << std::chrono::duration_cast<std::chrono::microseconds>(end - start).count() << " us\n";
        }
    }

    SpriteGroup::SpriteGroup(const std::string_view path, std::size_t count)
    {
        m_Entity = std::make_unique<Entity>();
        m_Entity->ID.resize(count);
        m_Entity->Transforms.resize(count);

        std::vector<Message::Packet> packets;
        packets.reserve(count);

        for (std::size_t i = 0; i < count; ++i)
        {
            auto packet = Message::Packets::CreateSprite{
                .Result = &m_Entity->ID[i],
                .Path   = path,
            };

            packets.push_back(std::move(packet));
        }

        Engine::GetInstance().Start();
        Engine::GetInstance().GetMessageQueue().EnqueueSync(std::move(packets));
    }
    
    SpriteGroup::~SpriteGroup()
    {
        if (!m_Entity)
        {
            return;
        }

        std::vector<Message::Packet> packets;
        packets.reserve(m_Entity->ID.size());

        for (std::size_t i = 0; i < m_Entity->ID.size(); ++i)
        {
            auto packet = Message::Packets::DestroyEntity{
                .ID = m_Entity->ID[i],
            };

            packets.push_back(std::move(packet));
        }

        Engine::GetInstance().GetMessageQueue().Enqueue(std::move(packets));
    }

    auto SpriteGroup::SetTransform(std::function<Transform&(Transform&)> delegate) -> void
    {
        if (!m_Entity)
        {
            return;
        }

        std::vector<Message::Packet> packets;
        packets.reserve(m_Entity->ID.size());

        for (std::size_t i = 0; i < m_Entity->ID.size(); ++i)
        {
            auto& transform = delegate(m_Entity->Transforms[i]);

            auto packet = Message::Packets::SetTransform{
                .ID = m_Entity->ID[i],
                .Transform = transform,
            };

            packets.push_back(std::move(packet));
        }

        Engine::GetInstance().GetMessageQueue().Enqueue(std::move(packets));
    }

        auto TranscodeUtf8ToWide(const std::string_view utf8) -> std::wstring
        {
            if (utf8.empty())
            {
                return {};
            }

            int desired = ::MultiByteToWideChar(CP_UTF8, 0, utf8.data(), -1, nullptr, 0);
            if (desired == 0)
            {
                return {};
            }

            std::wstring result(desired, 0);
            ::MultiByteToWideChar(CP_UTF8, 0, utf8.data(), -1, &result[0], desired);

            result.resize(desired - 1);
            return result;
        }
    } // namespace

    WicImageDecoder::WicImageDecoder(const std::string_view path) : m_Path(path)
    {
        auto factory = wil::CoCreateInstance<IWICImagingFactory>(CLSID_WICImagingFactory);

        THROW_IF_FAILED(factory->CreateDecoderFromFilename(TranscodeUtf8ToWide(path).c_str(), nullptr, GENERIC_READ, WICDecodeMetadataCacheOnDemand, &m_Decoder)

    auto WicImageDecoder::Decode(std::function<std::span<std::byte>(std::size_t)> allocate) -> Pixels::Buffer
    {
        // 逕ｻ蜒上・譛蛻昴・繝輔Ξ繝ｼ繝繧貞叙蠕励☆繧・
        wil::com_ptr<IWICBitmapFrameDecode> frame;
        THROW_IF_FAILED(m_Decoder->GetFrame(0, &frame));

        // 繝輔か繝ｼ繝槭ャ繝医さ繝ｳ繝舌・繧ｿ繝ｼ繧剃ｽ懈・縺励※縲√ヵ繝ｬ繝ｼ繝繧・2bppPBGRA蠖｢蠑上↓螟画鋤縺吶ｋ
        auto factory = wil::CoCreateInstance<IWICImagingFactory>(CLSID_WICImagingFactory);
        wil::com_ptr<IWICFormatConverter> converter;
        THROW_IF_FAILED(factory->CreateFormatConverter(&converter));

        // 32bppPBGRA蠖｢蠑上↓螟画鋤縺吶ｋ
        THROW_IF_FAILED(converter->Initialize(frame.get(), GUID_WICPixelFormat32bppPBGRA, WICBitmapDitherTypeNone, nullptr, 0.0f, WICBitmapPaletteTypeCustom));

        // 逕ｻ蜒上・蟷・・ｫ倥＆縲√ヴ繝・メ繧貞叙蠕励＠縺ｦ縲∝ｿ・ｦ√↑繝舌う繝域焚繧定ｨ育ｮ励☆繧・
        const std::uint32_t width          = GetWidth();
        const std::uint32_t height         = GetHeight();
        const std::uint32_t pitch          = GetPitch();
        const std::size_t totalBytesNeeded = static_cast<std::size_t>(pitch) * height;

        // 繝舌ャ繝輔ぃ繧貞牡繧雁ｽ薙※繧・
        auto target = allocate(totalBytesNeeded);

        if (target.empty() || target.size() < totalBytesNeeded)
        {
            return {};
        }

        // 螟画鋤縺輔ｌ縺溘ヴ繧ｯ繧ｻ繝ｫ繝・・繧ｿ繧偵ヰ繝・ヵ繧｡縺ｫ繧ｳ繝斐・縺吶ｋ
        THROW_IF_FAILED(converter->CopyPixels(nullptr, pitch, static_cast<UINT>(target.size()), reinterpret_cast<BYTE*>(target.data())));

        // 繝舌ャ繝輔ぃ繧定ｿ斐☆
        return {
            .Bytes  = target.data(),
            .Size   = totalBytesNeeded,
            .Width  = GetWidth(),
            .Height = GetHeight(),
            .Pitch  = pitch,
        };
    }

    auto DrawBitmap::Execute(Context& context) -> void
    {
        auto d2dContext = context.GetD2DContext();

        if (!Bitmap)
        {
            auto entry = Engine::GetInstance().GetResourceContainer().Get(ResourceHandle);

            if (!entry)
            {
                return;
            }

            d2dContext->CreateBitmap(
                D2D1::SizeU(entry->Pixels.Width, entry->Pixels.Height),
                entry->Pixels.Bytes,
                entry->Pixels.Pitch,
                D2D1::BitmapProperties1(D2D1_BITMAP_OPTIONS_NONE, D2D1::PixelFormat(DXGI_FORMAT_B8G8R8A8_UNORM, D2D1_ALPHA_MODE_PREMULTIPLIED)),
                &Bitmap
            );

            if (!Bitmap)
            {
                return;
            }
        }

        d2dContext->DrawBitmap(
            Bitmap.get(),
            D2D1::RectF(0.0f, 0.0f, static_cast<FLOAT>(Bitmap->GetPixelSize().width), static_cast<FLOAT>(Bitmap->GetPixelSize().height)),
            1.0f,
            D2D1_BITMAP_INTERPOLATION_MODE_LINEAR
        );
    }

    auto CommandList::Add(Command&& command) -> CommandHandle
    {
        // 遨ｺ縺阪せ繝ｭ繝・ヨ縺後↑縺・ｴ蜷医・辟｡蜉ｹ縺ｪ繝上Φ繝峨Ν繧定ｿ斐＠縺ｾ縺吶・
        if (m_AvailableHandles.empty())
        {
            return { .ID = Handle::CommandID::Invalid };
        }

        // 蛻ｩ逕ｨ蜿ｯ閭ｽ縺ｪ繝上Φ繝峨Ν縺ｮ繝ｪ繧ｹ繝医°繧画忰蟆ｾ繧貞叙蠕励＠縺ｾ縺吶・
        CommandHandle handle = m_AvailableHandles.back();
        m_AvailableHandles.pop_back();

        const auto index = static_cast<std::uint64_t>(handle.ID);

        m_Entries[index] = Entry{
            .Handle  = handle,
            .Command = std::move(command),
        };

        m_ActiveCount++;

        return handle;
    }

    auto CommandList::Remove(const CommandHandle handle) -> bool
    {
        const auto index = static_cast<std::uint64_t>(handle.ID);

        if (index >= MaxEntries)
        {
            return false;
        }

        // 荳紋ｻ｣繝√ぉ繝・け: 縺吶〒縺ｫ蜑企勁縺輔ｌ縺ｦ縺・ｋ縺九∝挨縺ｮ繧ｳ繝槭Φ繝峨↓蜀榊茜逕ｨ縺輔ｌ縺ｦ縺・ｋ蝣ｴ蜷医・辟｡隕悶＠縺ｾ縺吶・
        if (m_Entries[index].Handle.Generation != handle.Generation)
        {
            return false;
        }

        // 繧ｳ繝槭Φ繝峨ｒ遐ｴ譽・ｼ・td::monostate 縺ｸ謌ｻ縺呻ｼ・
        m_Entries[index].Command = std::monostate{};

        // 谺｡縺ｮ荳紋ｻ｣縺ｮ繝上Φ繝峨Ν繧剃ｽ懈・縺励※繝輔Μ繝ｼ繝ｪ繧ｹ繝医↓謌ｻ縺励∪縺吶・
        CommandHandle nextHandle = handle;
        nextHandle.Generation    = static_cast<Handle::Generation>(static_cast<std::uint64_t>(handle.Generation) + 1);

        m_AvailableHandles.push_back(nextHandle);
        m_ActiveCount--;

        return true;
    }

    auto CommandList::Execute(Context& context) -> bool
    {
        if (m_ActiveCount == 0)
        {
            return false;
        }

        for (auto& entry : m_Entries)
        {
            // std::monostate (index 0) 縺ｧ縺ｯ縺ｪ縺・怏蜉ｹ縺ｪ繧ｳ繝槭Φ繝峨・縺ｿ螳溯｡後＠縺ｾ縺吶・
            if (entry.Command.index() != 0)
            {
                std::visit(
                    [&context](auto& command)
                    {
                        // std::monostate 縺ｮ蝣ｴ蜷医・菴輔ｂ縺励↑縺・ｼ亥梛螳牙・縺ｮ縺溘ａ縺ｮ繧ｳ繝ｳ繝代う繝ｫ譎ょ・蟯撰ｼ・
                        if constexpr (!std::is_same_v<std::decay_t<decltype(command)>, std::monostate>)
                        {
                            command.Execute(context);
                        }
                    },
                    entry.Command
                );
            }
        }

        return true;
    }

    auto CommandList::Clear() -> void
    {
        m_ActiveCount = 0;
        m_Entries.fill({ .Command = std::monostate{} });
        m_AvailableHandles.clear();

        // 0逡ｪ縺九ｉ鬆・↓蛻晄悄繝上Φ繝峨Ν繧偵せ繧ｿ繝・け縺ｫ遨阪∩縺ｾ縺呻ｼ磯・・↓遨阪・縺薙→縺ｧ0逡ｪ縺九ｉ菴ｿ繧上ｌ繧九ｈ縺・↓縺励∪縺呻ｼ・
        for (std::int64_t i = MaxEntries - 1; i >= 0; --i)
        {
            m_AvailableHandles.push_back({
                .ID         = static_cast<Handle::CommandID>(i),
                .Type       = Handle::CommandType::Draw,
                .Generation = Handle::Generation::Default,
            });
        }
    }

    Context::Context()
    {

        D2D1_FACTORY_OPTIONS options{};

        THROW_IF_FAILED(::D2D1CreateFactory(D2D1_FACTORY_TYPE_SINGLE_THREADED, __uuidof(ID2D1Factory1), &options, m_D2DFactory.put_void()));

        THROW_IF_FAILED(::DWriteCreateFactory(DWRITE_FACTORY_TYPE_SHARED, __uuidof(IDWriteFactory), dwriteFactory.put()));

        THROW_IF_FAILED(::D3D11CreateDevice(
            nullptr,
            D3D_DRIVER_TYPE_HARDWARE,
            nullptr,
            creationFlags,
            featureLevels,
            ARRAYSIZE(featureLevels),
            D3D11_SDK_VERSION,
            d3d11Device.put(),
            nullptr,
            d3d11Context.put()
        ));

    auto Context::GetRenderTargetWindow() const noexcept -> HWND
    {
        wil::unique_event_nothrow createWindowCompletion;

        if (!createWindowCompletion.try_open(L"Local\\N503.CppWin32.Event.Core.CreateWindowCompletion", SYNCHRONIZE | EVENT_MODIFY_STATE))
        {
            return {};
        }

        if (::WaitForSingleObject(createWindowCompletion.get(), 0) != WAIT_OBJECT_0)
        {
            return {};
        }

        struct Context
        {
            HWND Result{};
            DWORD TargetPID{};
        };

        Context context{ nullptr, ::GetCurrentProcessId() };

        auto callback = [](HWND handle, LPARAM lParam) -> BOOL
        {
            auto pContext = reinterpret_cast<Context*>(lParam);

            DWORD pid{};
            ::GetWindowThreadProcessId(handle, &pid);

            if (pid == pContext->TargetPID)
            {
                wchar_t className[256]{};
                ::GetClassNameW(handle, className, 256);

                if (std::wcscmp(className, L"N503.CppWin32.Core.Window") == 0)
                {
                    pContext->Result = handle;
                    return FALSE;
                }
            }

            return TRUE;
        };

        ::EnumWindows(callback, reinterpret_cast<LPARAM>(&context));

        if (!context.Result)
        {
            return {};
        }
        else
        {
            // 謌仙粥縺励◆縺ｮ縺ｧ譌励ｒ荳九ｍ縺・
            createWindowCompletion.ResetEvent();
        }

        return context.Result;
    }

    {

        {

        Codec::WicImageDecoder decoder(Path);

        Pixels::Buffer pixels;
        pixels.Width  = decoder.GetWidth();
        pixels.Height = decoder.GetHeight();
        pixels.Pitch  = decoder.GetPitch();
        pixels.Bytes  = nullptr;
        pixels.Size   = static_cast<std::size_t>(pixels.Height) * pixels.Pitch;

        }

        *Result = handle;
        *Pixels = pixels;
    }

    auto Dispatcher::Dispatch(Context& context) -> void
    {
        auto envelopes = context.MessageQueue.PopAll();

        while (!envelopes.empty())
        {
            auto& message = envelopes.front();

            auto cleanup = wil::scope_exit(
                [&message]()
                {
                    if (message.Signal)
                    {
                        message.Signal->release();
                    }
                }
            );

            std::visit([&context](auto&& concrete) { concrete(context); }, message.Packet);

            envelopes.pop();
        }
    }

    auto Queue::Enqueue(Packet&& packet) -> void
    {
        bool isBusy = false;

#ifdef _DEBUG
        {
            const std::size_t currentSize = m_Buffer[m_BufferIndex].Container.size();
            const std::size_t capacity    = MaxMessageQueued; // 繝励・繝ｫ縺ｮ繧ｵ繧､繧ｺ

            if (currentSize >= capacity * 0.95) // 95%繧定ｶ・∴縺溘ｉ隴ｦ蜻・
            {
                ::OutputDebugStringW(L"EventQueue is congesting!\n");
            }
        }
#endif
        {
            const std::lock_guard lock{ m_Mutex };

            const std::size_t currentSize = m_Buffer[m_BufferIndex].Container.size();
            const std::size_t capacity    = MaxMessageQueued; // 繝励・繝ｫ縺ｮ繧ｵ繧､繧ｺ

            if (currentSize >= capacity * 0.95)
            {
                isBusy = true;
            }
            else
            {
                m_Buffer[m_BufferIndex].Container.push(Envelope{ .Packet = std::move(packet), .Signal = nullptr });
            }
        }

        if (isBusy)
        {
            EnqueueSync(std::move(packet));
        }
        else
        {
            m_WakeupEvent.SetEvent();
        }
    }

    auto Queue::EnqueueSync(Packet&& packet) -> void
    {
        std::binary_semaphore signal{ 0 };

        {
            const std::lock_guard lock{ m_Mutex };

            m_Buffer[m_BufferIndex].Container.push(Envelope{ .Packet = std::move(packet), .Signal = &signal });
        }

        m_WakeupEvent.SetEvent();

        signal.acquire();
    }

    /// @brief
    /// @return
    [[nodiscard]]
    auto Queue::PopAll() -> Container
    {
        const std::lock_guard lock{ m_Mutex };

        // UI繧ｹ繝ｬ繝・ラ縺ｧ貅懊ａ縺溘ョ繝ｼ繧ｿ繧偵窟llocator縺斐→縲榊･ｪ縺・悉繧・
        auto container = std::move(m_Buffer[m_BufferIndex].Container);

        // 遨ｺ縺ｫ縺ｪ縺｣縺溷ｴ謇縺ｫ縲∵ｭ｣縺励＞ Allocator・・torage縺ｸ縺ｮ繝昴う繝ｳ繧ｿ・峨ｒ蜀崎｣・｡ｫ
        m_Buffer[m_BufferIndex].Container = Container{ m_Buffer[m_BufferIndex].Allocator };

        // 谺｡縺ｮ Push 逕ｨ縺ｫ繧､繝ｳ繝・ャ繧ｯ繧ｹ繧貞・繧頑崛縺医ｋ
        m_BufferIndex = (m_BufferIndex + 1) % m_Buffer.size();

        return container;
    }

    /// @return
    [[nodiscard]]
    auto Queue::GetWakeupEventHandle() const -> HANDLE
    {
        return m_WakeupEvent.get();

namespace N503::Renderer2D::Message
{

    class Queue final
    {
        static const std::size_t MaxMessageQueued = 4096;

        static const std::size_t BufferCount = 2;

    public:
        using Element = Envelope;

        struct Slot
        {
            alignas(void*) std::byte data[sizeof(Element) + 32];
        };

        using Storage = ::N503::Memory::Storage::Pool<Slot>;

        using Allocator = ::N503::Memory::StorageAllocator<Element, Storage>;

        using Strategy = std::list<Element, Allocator>;

        using Container = std::queue<Element, Strategy>;

    public:
        template <typename TPacket> class MessageBorrower
        {
        public:
            MessageBorrower(std::unique_lock<std::mutex>&& lock, Container& queue) : m_Lock(std::move(lock))
            {
                if (!queue.empty() && std::holds_alternative<TPacket>(queue.back().Packet))
                {
                    m_Target = &std::get_if<TPacket>(queue.back().Packet);
                }
            }

            MessageBorrower(const MessageBorrower&) = delete;

            auto operator=(const MessageBorrower&) -> MessageBorrower& = delete;

            auto operator->() -> TPacket*
            {
                return m_Target;
            }

            explicit operator bool() const
            {
                return m_Target != nullptr;
            }

        private:
            std::unique_lock<std::mutex> m_Lock;

            TPacket* m_Target = nullptr;
        };

    public:
        Queue() = default;

        Queue(const Queue&) = delete;

        auto operator=(const Queue&) -> Queue& = delete;

        Queue(Queue&&) = delete;

        auto operator=(Queue&&) -> Queue& = delete;

    public:
        auto Enqueue(Packet&&) -> void;

        auto EnqueueSync(Packet&&) -> void;

        [[nodiscard]]
        auto PopAll() -> Container;

        [[nodiscard]]
        auto GetWakeupEventHandle() const -> HANDLE;

    public:
        template <typename TDataType> auto TryBorrowBack() -> MessageBorrower<TDataType>
        {
            // 繝ｭ繝・け繧貞･ｪ蜿悶☆繧・
            std::unique_lock<std::mutex> lock{ m_Mutex };

            // 繝ｭ繝・け縺励◆迥ｶ諷九〒 Container 繧堤音螳壹＠縲〕ock 縺斐→ Borrower 縺ｫ貂｡縺・
            return MessageBorrower<TDataType>{ std::move(lock), m_Buffer[m_BufferIndex].Container };
        }

    private:
        /// @brief

    auto Container::Add(const std::string_view path, Pixels::Buffer& pixels) -> ResourceHandle
    {
        // 縺吶〒縺ｫ蜷後§繝代せ縺ｮ繝ｪ繧ｽ繝ｼ繧ｹ縺悟ｭ伜惠縺吶ｋ蝣ｴ蜷医・縲√◎縺ｮ繝上Φ繝峨Ν繧定ｿ斐＠縺ｾ縺吶・
        if (auto it = m_Indexes.find(path); it != m_Indexes.end())
        {
            return it->second;
        }

        // 蛻ｩ逕ｨ蜿ｯ閭ｽ縺ｪ繝上Φ繝峨Ν縺後↑縺・ｴ蜷医・縲∫┌蜉ｹ縺ｪ繝上Φ繝峨Ν繧定ｿ斐＠縺ｾ縺吶・
        if (m_AvailableHandles.empty())
        {
            return { .ID = Handle::ResourceID::Invalid };
        }

        // 蛻ｩ逕ｨ蜿ｯ閭ｽ縺ｪ繝上Φ繝峨Ν縺ｮ繝ｪ繧ｹ繝医°繧峨ワ繝ｳ繝峨Ν繧貞叙蠕励＠縺ｾ縺吶・
        ResourceHandle handle = m_AvailableHandles.back();
        m_AvailableHandles.pop_back();

        // 繝上Φ繝峨Ν縺ｮID繧偵う繝ｳ繝・ャ繧ｯ繧ｹ縺ｨ縺励※菴ｿ逕ｨ縺励∪縺吶ゅ％繧後↓繧医ｊ縲＾(1)縺ｧ繧ｨ繝ｳ繝医Μ縺ｫ繧｢繧ｯ繧ｻ繧ｹ縺ｧ縺阪∪縺吶・
        const auto index = static_cast<std::uint64_t>(handle.ID);

        // 逕ｻ蜒上ョ繝ｼ繧ｿ縺ｮ繝舌う繝医し繧､繧ｺ縺ｫ蝓ｺ縺･縺・※縲√い繝ｪ繝｡繝ｳ繝医ｒ16繝舌う繝医→縺励◆逕溘・繝舌う繝亥・縺ｨ縺励※繝｡繝｢繝ｪ繧堤｢ｺ菫昴＠縺ｾ縺吶・
        void* address = m_Storage.AllocateBytes(pixels.Size, 16);

        // 蜻ｼ縺ｳ蜃ｺ縺怜・縺ｫ譖ｸ縺崎ｾｼ縺ｿ蜿ｯ閭ｽ縺ｪ繝舌ャ繝輔ぃ縺ｮ菴咲ｽｮ繧剃ｼ昴∴繧九◆繧√￣ixels讒矩菴薙・Bytes縺ｫ繧｢繝峨Ξ繧ｹ繧偵そ繝・ヨ縺励∪縺吶・
        pixels.Bytes = static_cast<std::byte*>(address);

        // 繧ｨ繝ｳ繝医Μ繧剃ｽ懈・縺励※菫晏ｭ倥＠縺ｾ縺吶・
        m_Entries[index] = Entry{
            .Handle   = handle,
            .Pixels   = pixels,
            .Metadata = { .Path = std::string(path) },
        };

        // 繝代せ縺ｨ繝上Φ繝峨Ν縺ｮ蟇ｾ蠢懊ｒ菫晏ｭ倥＠縺ｾ縺吶・
        m_Indexes[m_Entries[index].Metadata.Path] = handle;

        return handle;
    }

    auto Container::Remove(const ResourceHandle handle) -> bool
    {
        // 繝上Φ繝峨Ν縺ｮID繧偵う繝ｳ繝・ャ繧ｯ繧ｹ縺ｨ縺励※菴ｿ逕ｨ縺励∪縺吶・
        const auto index = static_cast<std::uint64_t>(handle.ID);

        if (index >= MaxEntries)
        {
            return false;
        }

        // 繝上Φ繝峨Ν縺ｮ荳紋ｻ｣縺後お繝ｳ繝医Μ縺ｮ荳紋ｻ｣縺ｨ荳閾ｴ縺励↑縺・ｴ蜷医・縲√☆縺ｧ縺ｫ辟｡蜉ｹ縺ｧ縺ゅｋ縺九√≠繧九＞縺ｯ荳紋ｻ｣縺悟商縺・→縺ｿ縺ｪ縺励∪縺吶・
        if (m_Entries[index].Handle.Generation != handle.Generation)
        {
            return false; // 縺吶〒縺ｫ辟｡蜉ｹ縲√≠繧九＞縺ｯ荳紋ｻ｣縺悟商縺・
        }

        // 繝代せ縺ｨ繝上Φ繝峨Ν縺ｮ蟇ｾ蠢懊ｒ蜑企勁縺励∪縺吶・
        m_Indexes.erase(m_Entries[index].Metadata.Path);

        // 繧ｨ繝ｳ繝医Μ繧堤ｩｺ縺ｫ縺励※縲∵ｬ｡縺ｮ荳紋ｻ｣縺ｮ繝上Φ繝峨Ν繧貞茜逕ｨ蜿ｯ閭ｽ縺ｪ繝上Φ繝峨Ν縺ｮ繝ｪ繧ｹ繝医↓霑ｽ蜉縺励∪縺吶・
        ResourceHandle nextGenerationHandle = handle;
        nextGenerationHandle.Generation     = static_cast<Handle::Generation>(static_cast<std::uint64_t>(handle.Generation) + 1);

        // 繧ｨ繝ｳ繝医Μ繧堤ｩｺ縺ｫ縺励∪縺吶・
        m_Entries[index] = {};

        // 谺｡縺ｮ荳紋ｻ｣縺ｮ繝上Φ繝峨Ν繧貞茜逕ｨ蜿ｯ閭ｽ縺ｪ繝上Φ繝峨Ν縺ｮ繝ｪ繧ｹ繝医↓霑ｽ蜉縺励∪縺吶・
        m_AvailableHandles.push_back(nextGenerationHandle);

        return true;
    }

    auto Container::Get(const ResourceHandle handle) const -> const Entry*
    {
        const auto index = static_cast<std::uint64_t>(handle.ID);

        if (index >= MaxEntries)
        {
            return nullptr;
        }

        if (m_Entries[index].Handle.Generation != handle.Generation)
        {
            return nullptr; // 辟｡蜉ｹ縲√≠繧九＞縺ｯ荳紋ｻ｣縺悟商縺・
        }

        return &m_Entries[index];
    }

    auto Container::Get(const std::string_view path) const -> const Entry*
    {
        if (auto it = m_Indexes.find(path); it != m_Indexes.end())
        {
            return Get(it->second);
        }

        return nullptr;
    }

    auto Container::GetHandle(const std::string_view path) const -> ResourceHandle
    {
        if (auto it = m_Indexes.find(path); it != m_Indexes.end())
        {
            return it->second;
        }

        return { .ID = Handle::ResourceID::Invalid };
    }

    auto Container::Clear() -> void
    {
        // 縺吶∋縺ｦ縺ｮ繧ｨ繝ｳ繝医Μ繧偵け繝ｪ繧｢縺励∝茜逕ｨ蜿ｯ閭ｽ縺ｪ繝上Φ繝峨Ν縺ｮ繝ｪ繧ｹ繝医ｒ蛻晄悄蛹悶＠縺ｾ縺吶・
        m_Indexes.clear();
        m_Entries.fill({});
        m_AvailableHandles.clear();

        // Arena縺ｮ迥ｶ諷九ｒ繝ｪ繧ｻ繝・ヨ縺励∪縺吶ゅ％繧後↓繧医ｊ縲√☆縺ｹ縺ｦ縺ｮ遒ｺ菫昴＆繧後◆繝｡繝｢繝ｪ縺瑚ｧ｣謾ｾ縺輔ｌ縲∵ｬ｡縺ｮ遒ｺ菫昴・譛蛻昴・繝悶Ο繝・け縺九ｉ蟋九∪繧翫∪縺吶・
        m_Storage.Reset();

        // 蛻ｩ逕ｨ蜿ｯ閭ｽ縺ｪ繝上Φ繝峨Ν縺ｮ繝ｪ繧ｹ繝医ｒ蛻晄悄蛹悶＠縺ｾ縺吶ゅ％繧後↓繧医ｊ縲∵怙蛻昴・MaxEntries蛟九・繝ｪ繧ｽ繝ｼ繧ｹID縺悟茜逕ｨ蜿ｯ閭ｽ縺ｫ縺ｪ繧翫∪縺吶・
        for (std::uint64_t i = 0; i < MaxEntries; ++i)
        {
            m_AvailableHandles.push_back({ .ID = static_cast<Handle::ResourceID>(i), .Generation = Handle::Generation::Default });
        }
    }

    auto Engine::Start() -> void
    {
        // 縺吶〒縺ｫ繧ｨ繝ｳ繧ｸ繝ｳ縺碁幕蟋九＆繧後※縺・ｋ蝣ｴ蜷医・縲∽ｽ輔ｂ縺励↑縺・
        if (m_IsRunning.load(std::memory_order_acquire))

            return;
        }

        std::binary_semaphore signal{ 0 };

        // 繝ｬ繝ｳ繝繝ｩ繝ｼ 繧ｹ繝ｬ繝・ラ繧剃ｽ懈・縺吶ｋ
        m_RendererThread = std::jthread(
            [this, &signal](const std::stop_token stopToken)
            {
                // 繧ｹ繝ｬ繝・ラ蜷阪ｒ險ｭ螳壹☆繧・
                ::SetThreadDescription(::GetCurrentThread(), L"N503.CppWin32.Renderer2D");

                // 繝ｬ繝ｳ繝繝ｩ繝ｼ 繧ｹ繝ｬ繝・ラ縺ｮ蛻晄悄蛹悶′螳御ｺ・＠縺溘％縺ｨ繧帝夂衍縺吶ｋ
                signal.release();

                // 繝ｬ繝ｳ繝繝ｩ繝ｼ 繧ｹ繝ｬ繝・ラ縺ｮ蛻晄悄蛹悶′螳御ｺ・＠縺溘％縺ｨ繧堤､ｺ縺吶う繝吶Φ繝医ｒ繧ｷ繧ｰ繝翫Ν迥ｶ諷九↓縺吶ｋ
                m_StartedEvent.SetEvent();

                // COM 繧貞・譛溷喧縺吶ｋ
                auto coinit = wil::CoInitializeEx(COINIT_MULTITHREADED);

                // PeekMessageW繧貞他縺ｳ蜃ｺ縺励※縲√Ξ繝ｳ繝繝ｩ繝ｼ繧ｹ繝ｬ繝・ラ縺ｮ繝｡繝・そ繝ｼ繧ｸ繧ｭ繝･繝ｼ繧剃ｽ懈・縺吶ｋ
                MSG msg{};
                ::PeekMessageW(&msg, nullptr, WM_USER, WM_USER, PM_NOREMOVE);

                // 繝ｬ繝ｳ繝繝ｩ繝ｼ繧ｹ繝ｬ繝・ラ繧帝幕蟋九☆繧・
                Run(std::move(stopToken));

                // 繝ｬ繝ｳ繝繝ｩ繝ｼ 繧ｹ繝ｬ繝・ラ縺悟●豁｢縺励◆縺薙→繧堤､ｺ縺吶う繝吶Φ繝医ｒ繝ｪ繧ｻ繝・ヨ迥ｶ諷九↓縺吶ｋ
                m_IsRunning.store(false, std::memory_order_release);
            }
        );

        // 繧ｹ繝ｬ繝・ラ縺瑚ｵｷ蜍輔＠縺ｦ蛻晄悄蛹悶′螳御ｺ・☆繧九∪縺ｧ蠕・ｩ溘☆繧・
        signal.acquire();

        // 繧ｨ繝ｳ繧ｸ繝ｳ縺碁幕蟋九＆繧後◆縺薙→繧堤､ｺ縺吶う繝吶Φ繝医ｒ繧ｷ繧ｰ繝翫Ν迥ｶ諷九↓縺吶ｋ
        m_IsRunning.store(true, std::memory_order_release);

    auto Engine::Stop() -> void
    {
        if (!::PostThreadMessage(::GetThreadId(m_RendererThread.native_handle()), WM_QUIT, 0, 0))
        {
            const Diagnostics::Entry entry{
                .Severity = Diagnostics::Severity::Error,
                .Expected = std::format("PostThreadMessage failed: Reason={}, Handle={}\n", ::GetLastError(), m_RendererThread.native_handle()).data(),
                .Position = 0
            };

            m_DiagnosticsSink.AddEntry(entry);
        }
    }

    auto Engine::Wait() -> void
    {

        }
    }

    auto Engine::Run(const std::stop_token stopToken) -> void
    {
        std::unique_ptr<Device::Context> deviceContext     = std::make_unique<Device::Context>();
        std::unique_ptr<Device::RenderTarget> renderTarget = nullptr;
        std::unique_ptr<Device::CommandList> commandList   = std::make_unique<Device::CommandList>();

        // 繝｡繝・そ繝ｼ繧ｸ 繝・ぅ繧ｹ繝代ャ繝√Ε繝ｼ繧貞・譛溷喧縺吶ｋ
        Message::Dispatcher messageDispatcher;

        // 繝ｬ繝ｳ繝繝ｩ繝ｼ繧ｹ繝ｬ繝・ラ縺ｮ邨ゆｺ・凾縺ｫ繝ｪ繧ｽ繝ｼ繧ｹ繧偵け繝ｪ繝ｼ繝ｳ繧｢繝・・縺吶ｋ縺溘ａ縺ｮ繧ｹ繧ｳ繝ｼ繝・繧ｬ繝ｼ繝・
        auto cleanupResources = wil::scope_exit(
            [&]
            {
                renderTarget.reset();
                deviceContext.reset();
                commandList.reset();
            }
        );

        // OS 繝｡繝・そ繝ｼ繧ｸ繧偵ョ繧｣繧ｹ繝代ャ繝√☆繧九Λ繝繝蠑・
        auto OSMessageDispatch = []() -> bool
        {
            MSG message{};

            while (::PeekMessage(&message, nullptr, 0, 0, PM_REMOVE))
            {
                if (message.message == WM_QUIT)
                {
                    return false;
                }

                ::TranslateMessage(&message);
                ::DispatchMessageW(&message);
            }

            return true;
        };

        // 繝ｬ繝ｳ繝繝ｩ繝ｼ 繧ｹ繝ｬ繝・ラ縺ｮ繝｡繧､繝ｳ 繝ｫ繝ｼ繝・
        auto waitHandles = { m_MessageQueue->GetWakeupEventHandle() };
        auto isAnyActive = false;

        while (!stopToken.stop_requested())
        {
            HWND hwnd{};

            if (renderTarget == nullptr && (hwnd = deviceContext->GetRenderTargetWindow()))
            {
                renderTarget = std::make_unique<Device::RenderTarget>(deviceContext.get(), hwnd);
                deviceContext->SetTarget(*renderTarget);
            }

            const auto count   = static_cast<DWORD>(waitHandles.size());
            const auto handles = waitHandles.begin();
            const auto timeout = isAnyActive ? 0 : INFINITE;

            auto result = ::MsgWaitForMultipleObjectsEx(count, handles, timeout, QS_ALLINPUT, MWMO_INPUTAVAILABLE);

            if (result >= WAIT_OBJECT_0 && result < (WAIT_OBJECT_0 + count))
            {
                Message::Context context{
                    .MessageQueue      = *m_MessageQueue,
                    .DeviceContext     = *deviceContext,
                    .CommandList       = *commandList,
                    .ResourceContainer = *m_ResourceContainer,
                };

                messageDispatcher.Dispatch(context);
            }
            else if (result == WAIT_OBJECT_0 + count)
            {
                if (!OSMessageDispatch())
                {
                    return;
                }
            }

            if (deviceContext->BeginDraw())
            {
                isAnyActive = commandList->Execute(*deviceContext);

                const auto endDrawResult = deviceContext->EndDraw();
                const auto presentResult = renderTarget->Present();

                if (endDrawResult == D2DERR_RECREATE_TARGET || presentResult == DXGI_ERROR_DEVICE_REMOVED || presentResult == DXGI_ERROR_DEVICE_RESET)
                {
                    // 繝・ヰ繧､繧ｹ繝ｭ繧ｹ繝育匱逕溘ゅΜ繧ｽ繝ｼ繧ｹ繧偵け繝ｪ繝ｼ繝ｳ繧｢繝・・縺励※繧ｳ繝ｳ繝・く繧ｹ繝医ｒ蜀肴ｧ狗ｯ峨☆繧・
                    deviceContext = std::make_unique<Device::Context>();
                    renderTarget  = std::make_unique<Device::RenderTarget>(deviceContext.get(), renderTarget->GetTargetWindow());
                    deviceContext->SetTarget(*renderTarget);
                }
            }
        }
    }
